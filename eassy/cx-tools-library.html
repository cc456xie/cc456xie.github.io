# 工具函数库文档

# 安装

npm i cx-tools-library

# 导入

  • 全局引入
const X = require('cx-tools-library')

function fn = X.xfp.curry()
  • 按需引入
const {xArray} = require('cx-tools-library')

这里只引用了数组处理的方法

# 数组

  • 引入数组模块
const {xArray} = require('cx-tools-library')
  • 升序排序
let arr = [3,2,1]

console.log( xArray.upSort(arr)) //[1,2,3]
  • 降序排序
let arr = [1,2,3]

console.log( xArray.downSort(arr)) //[3,2,1]
  • 乱序
let arr = [1,2,3,4,5]

console.log( xArray.disorder(arr)) //[ 2, 1, 4, 5, 3 ]
  • 交换数组下标元素
let arr = [1,2,3,4,5]
 xArray.swap(arr,0,3)
console.log(arr) //[ 5,2,3,4,1 ]
  • 数组去重
let arr = [1,2,2,3,1]

console.log(xArray.disRepeat(arr)) //[1,2,3]
  • 数组反转
let arr = [1,2,3]
console.log(xArray.reverse(arr)) //[3,2,1]

# 日期

  • 引入日期模块
const {xDate} = require('cx-tools-library')
  • 时间戳转化日期

请使用ms时间戳(s时间戳需*1000)

let date = 1592717613278

let newDate = xDate.conversion(date)

console.log(newDate) // 2020-06-21 13:33
  • 计算两日期之间相差的天数 参数为时间戳
let  date1 = 1590039200000,date2 = 1592717613278

let differ = xDate.differDate(date1,date2)

console.log(`相差${differ}天`)  //相差31天
  • 计算某个日期与当前日期相差的天数
let  date = 1590039200000

let differ = xDate.differNow(date)

console.log(`相差${differ}天`)  //相差31天

# 函数式编程

  • 引入该模块
const {xfp} = require('cx-tools-library')
  • 柯里化
    let sum =function(a,b,c)
   {
   return a + b + c
   }   

    const sumCurry = xfp.curry(sum)

    let sumer =  sumCurry(1)(2)(3) 
    console.log(sumer)   //6
  • 组合函数
   let fn1 =function(str)
   {
   return str.toUpperCase()

   }  
    let fn2 =function(str)
   {
   return str.split('').reverse().join('')
   }
    let fn3 =function(str)
   {
   return str[0]
   } 

    let fn = X.xfp.compose(fn3,fn2,fn1)

    console.log(fn('asd'))   //D

上图的三个函数组合成一个组合函数，功能是将一个字符串反转，然后转为大写，然后取出第一个字符

# 随机数

  • 引入随机数模块
const {xRandom} = require('cx-tools-library')
  • 生成带权重的随机数
let weightArr = [{data:0,weight:1},{data:1,weight:8},{data:2,weight:1}]

for(let i=0;i<10;i++)
{
    console.log(xRandom.wRandom(weightArr)) // 1,1,1,1,1,1,1,0,1,2
}

weight值不宜过大，否则会有性能问题

  • 生成规定范围的随机数
for(let i=0;i<5;i++)e
{
    console.log(xRandom.eRandom(10,20)) // 19,17,15,10,13
    console.log(xRandom.eRandom(20)) // 2,5,19,4,18
}

若不传第二个参数，默认从零开始

  • 生成随机数组
//表示生成长度为五，值为10~20的随机数组
console.log(xRandom.randoms(5,10,20)) // [ 11, 11, 10, 13, 19 ]

# 正则匹配

  • 引入匹配模块
const {xReg} = require('cx-tools-library')
  • 邮箱匹配
let email = '1109044172@qq.com'
console.log(xReg.emailReg(email)) //true
console.log(xReg.emailReg('123')) //false
  • URL匹配
let url = 'https://www.bilibili.com/'
console.log(xReg.emailReg(url)) //true
console.log(xReg.emailReg('123')) //false
