# 防抖

		//防抖
		function debounce(fn,wait,immediate) {
			let timer = null
			return function(){	
				timer && clearTimeout(timer) //清除定时器
				if(immediate){ //若立即执行
					let doNow = !timer //若timer存在，则说明还没完成时间间隔，则不触发函数
					timer = setTimeout(() => {
						timer = null  //若定时完成，则清除timer，下次触发时doNow即可为true，函数可立即执行
					},wait)
					doNow && fn.apply(this,arguments)
				}else{
					timer = setTimeout(() => {
						fn.apply(this,arguments) //出发函数后等待时间后会执行fn，若在等待期间再次出发函数，会重置定时器
					},wait)
				}
			}
		}

# 时间戳版

		//节流
		//时间戳实现
		function throttle1(fn,wait){
			let old = 0 //初始时间戳为0，这样就可立即执行
			return function(...args){
				let now = new Date().valueOf() //获取当前时间戳
				if( now - old > wait)
				{
					fn.call(this,args)
					old = now //更新初始时间戳，开始进入下次等待
				}
			}
		}

# 定时器版

		
		//定时器实现
		function throttle2(fn,wait){
			let timer = null
			return function(...args){
				if(!timer)
				{
					timer = setTimeout(() => {
						timer = null
						fn.call(this,args)
					}, wait);				
			}
		}
		
		}

# 结合版

时间戳加定时器实现 由于时间戳不能在最后触发时延迟打印，定时器不能在开头立即打印，所以结合两者优点

		function throttle(fn,wait){
			let timer = null,old = 0
			return function(){
			let now = new Date().valueOf()
				if( now - old >= wait )
				{	
				
					if(timer)
					 {
						clearTimeout(timer) 
						 timer = null	
					 }
					old = now
					fn.apply(this,arguments)	
					console.log('a')
				}
				
				if(!timer)
				{					
					timer = setTimeout(() => {
						old = new Date().valueOf()
						fn.apply(this,arguments)
						timer = null
						console.log('b')
					}, wait);				
				}
			}
		}
